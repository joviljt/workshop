from flask import Flask, render_template_string, request

app = Flask(__name__)

# Embedded HTML with Bootstrap (mobile-first design)
HTML_TEMPLATE = """
<!doctype html>
<html lang="en">
<head>
  <meta charset="utf-8">
  <meta name="viewport" content="width=device-width, initial-scale=1">
  <title>Simple Mobile Flask</title>
  <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/css/bootstrap.min.css" rel="stylesheet">
</head>
<body class="bg-light">
  <div class="container py-4">
    <h1 class="mb-4">ðŸ‘‹ Hello from Flask (Mobile Web)</h1>

    <form method="post" class="mb-3">
      <label for="name" class="form-label">Your Name:</label>
      <input type="text" name="name" id="name" class="form-control" placeholder="Enter your name" required>
      <button type="submit" class="btn btn-primary mt-3">Submit</button>
    </form>

    {% raw %}{% if name %}{% endraw %}
      <div class="alert alert-success">Hi, <strong>{{ name }}</strong>! Welcome to Flask ðŸ˜Š</div>
    {% raw %}{% endif %}{% endraw %}
  </div>
</body>
</html>
"""

@app.route("/", methods=["GET", "POST"])
def index():
    name = request.form.get("name") if request.method == "POST" else None
    return render_template_string(HTML_TEMPLATE, name=name)

if __name__ == "__main__":
    app.run(debug=True, host="0.0.0.0", port=5000)
